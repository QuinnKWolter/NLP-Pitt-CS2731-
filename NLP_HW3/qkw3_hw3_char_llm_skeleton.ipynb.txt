    "    model = model.to(device)  # Move model to the correct device at function start... \"CUDA assert\" errors otherwise.\n",
    "    model.eval()\n",
    "    total_log_likelihood = 0.0\n",
    "    total_tokens = 0\n",
    "\n",
    "    with torch.no_grad():  \n",
    "        input_ids = tokenizer.encode(text, return_tensors=\"pt\").to(device) # move them here?\n",
    "\n",
    "        # Process in chunks\n",
    "        for i in range(0, input_ids.shape[1], stride):\n",
    "            chunk = input_ids[:, i : i + max_length]\n",
    "            if chunk.shape[1] < 2:  # Skip empty sequences\n",
    "                continue\n",
    "\n",
    "            outputs = model(chunk, labels=chunk)\n",
    "            loss = outputs.loss.item()\n",
    "\n",

    "            total_tokens += chunk.shape[1]\n",
